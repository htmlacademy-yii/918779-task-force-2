class IsCityAvailable {
  Проверяет существует ли город в таблице городов
}

class Role {
  Проверяет какая роль у данного пользователя
}

class User {
  Сохраняет данные пользователя полученные из формы регистарации
}

class IsCategoryAvailable {
  Проверяет существует ли категория в таблице категорий
}

class Files {
  Переносит сохраненные файлы из формы добавления задания в публичную дирректорию
  и сохраняет ссылки на них
}

class Executor {
  Сохраняет или изменяет данные Исполнителя, записывает их в таблицу, передает данные об Исполнителе
  из таблицы на страницу Профиля
}

class Response {
  Сохраняет отклик в таблицу с откликами и выводит отклик на страницу Задания
}

class Review {
  Записывает отзыв в таблицу с отзывами, выводит отзыв на страницу Исполнителя, меняет данные Исполнителя в таблице исполнителей - количество выполненных заданий,
  среднюю оценку. Считает текущую статистику пользователя и выводит его рейтинг и статистику на страницу профиль Исполнителя
}
